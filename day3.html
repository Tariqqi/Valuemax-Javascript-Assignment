Drink Recommendation


function runCode() {
    let age = prompt('How old are you?');
    let gender = prompt('Male or female?');
    let message = '';

    if (age < 18) {
        message = 'You can take only water';
    } else if (age > 18 && age < 25 && gender.toLowerCase() === 'female') {
        message = 'You can take coke';
    } else if (age > 25 && gender.toLowerCase() === 'female') {
        message = 'You may take beer';
    } else if (age > 18 && gender.toLowerCase() === 'male') {
        message = 'You may take beer';
    } else {
        message = 'No drink for you';
    }

    document.getElementById('output').innerHTML = `
${message}
`;
}
